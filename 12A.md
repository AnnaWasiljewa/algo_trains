### Возрастает ли список? 

Дан список. Определите, является ли он монотонно возрастающим(то есть верно ли, 
что каждый элемент этого списка больше предыдущего).
Выведите _YES_, если массив монотонно возрастает и _NO_ в противном случае.

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
	vector<int>arr;
	int num;
	
	while(std::cin >> num) {
	    arr.push_back(num);
	}
	
	if (arr.size() <= 1) {
	    cout << "YES" << endl;
	    return 0;
	}
	
	bool is_increasing = true;
	for (size_t i = 1; i < arr.size(); ++i) {
	    if (arr[i] <= arr[i-1]) {
	        is_increasing = false;
	        break;
	    }
	}
	
	if (is_increasing) {
	    cout << "YES" << endl;
	} else {
	    cout << "NO" << endl;
	}
	
	return 0;
}
```
Время: 4ms. Память: 3.10Mb.

### Самое частое слово

Дан текст. Выведите слово, которое в этом тексте встречается чаще всего (с учетом регистра, 
и следующих знаков пунктуации за ним – не удаляйте их, и не приводите строку к нижнему регистру). 
Если таких слов несколько, выведите то, которое меньше в лексикографическом порядке.

```cpp
#include <iostream>
#include <sstream>
#include <string>
#include <map>

int main() {
    std::string text;
    
    // Читаем весь ввод до конца (может быть несколько строк)
    std::ostringstream oss;
    while (std::getline(std::cin, text) && !text.empty()) {
        oss << text << " ";
    }
    std::string fullText = oss.str();
    
    std::istringstream iss(fullText);
    std::string word;
    std::map<std::string, int> wordCount;
    
    // Разбиваем текст на слова и подсчитываем их частоту
    while (iss >> word) {
        wordCount[word]++;
    }
    
    std::string mostFrequentWord;
    bool firstWord = true;
    int maxCount = 0;
    
    // Ищем слово с максимальной частотой, выбирая лексикографически наименьшее при равенстве частот
    for (const auto& pair : wordCount) {
        const std::string& currentWord = pair.first;
        int currentCount = pair.second;
        
        if (firstWord) {
            mostFrequentWord = currentWord;
            maxCount = currentCount;
            firstWord = false;
        } else if (currentCount > maxCount) {
            mostFrequentWord = currentWord;
            maxCount = currentCount;
        } else if (currentCount == maxCount && currentWord < mostFrequentWord) {
            mostFrequentWord = currentWord;
        }
    }
    
    // Выводим результат
    std::cout << mostFrequentWord << std::endl;
    
    return 0;
}
```

### Наибольшее произведение двух чисел

Дан список, заполненный произвольными целыми числами. 
Найдите в этом списке два числа, произведение которых максимально. Выведите эти числа в порядке неубывания.
Список содержит не менее двух элементов. Числа подобраны так, что ответ однозначен.
Решение должно иметь сложность O(n), где _n_ - размер списка.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    vector<int> numbers;
    int num;

    // Считываем все числа до конца ввода
    while (cin >> num) {
        numbers.push_back(num);
    }

    // Инициализируем переменные первыми двумя элементами
    int max1, max2; // Два наибольших числа
    int min1, min2; // Два наименьших числа

    if (numbers[0] > numbers[1]) {
        max1 = numbers[0];
        max2 = numbers[1];
        min1 = numbers[1];
        min2 = numbers[0];
    } else {
        max1 = numbers[1];
        max2 = numbers[0];
        min1 = numbers[0];
        min2 = numbers[1];
    }

    // Проходим по оставшимся элементам (с индекса 2 до конца)
    for (size_t i = 2; i < numbers.size(); ++i) {
        int current = numbers[i];

        // Обновляем два наибольших
        if (current > max1) {
            max2 = max1;
            max1 = current;
        } else if (current > max2) {
            max2 = current;
        }

        // Обновляем два наименьших
        if (current < min1) {
            min2 = min1;
            min1 = current;
        } else if (current < min2) {
            min2 = current;
        }
    }

    // Считаем произведения (используем long long для предотвращения переполнения)
    long long product_max = (long long)max1 * max2;
    long long product_min = (long long)min1 * min2;

    // Выбираем пару с максимальным произведением
    int a, b;
    if (product_max > product_min) {
        a = max2;
        b = max1;
    } else {
        a = min1;
        b = min2;
    }

    // Гарантируем порядок неубывания: a <= b
    if (a > b) {
        swap(a, b);
    }

    // Выводим строго по формату: два числа через пробел, перевод строки в конце
    cout << a << " " << b << endl;

    return 0;
}
```
Время: 35ms. Память: 796Kb.

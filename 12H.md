### Наибольшее прозведение трёх чисел

В данном списке из _n_ ≤ $10^5$ целых чисел найдите три числа,произведение которых максимально.
Решение должно иметь сложность O(n), где _n_ - размер списка. Выведите три искомых числа в любом порядке.

```cpp
#include <iostream>
#include <vector>
#include <climits>

using namespace std;

int main() {
    vector<int> numbers;
    int num;

    // Считываем все числа до конца ввода
    while (cin >> num) {
        numbers.push_back(num);
    }

    // Инициализируем переменные
    long long max1 = LLONG_MIN, max2 = LLONG_MIN, max3 = LLONG_MIN;
    long long min1 = LLONG_MAX, min2 = LLONG_MAX;

    // Один проход по массиву для нахождения нужных значений
    for (int current : numbers) {
        // Обновляем три наибольших
        if (current > max1) {
            max3 = max2;
            max2 = max1;
            max1 = current;
        } else if (current > max2) {
            max3 = max2;
            max2 = current;
        } else if (current > max3) {
            max3 = current;
        }

        // Обновляем два наименьших
        if (current < min1) {
            min2 = min1;
            min1 = current;
        } else if (current < min2) {
            min2 = current;
        }
    }

    // Считаем два возможных максимальных произведения
    long long product1 = max1 * max2 * max3;
    long long product2 = min1 * min2 * max1;

    // Выводим три числа с максимальным произведением в любом порядке
    if (product1 >= product2) {
        cout << max1 << " " << max2 << " " << max3 << endl;
    } else {
        cout << min1 << " " << min2 << " " << max1 << endl;
    }

    return 0;
}
```
Время: 37ms. Память: 792Kb.
